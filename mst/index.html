Kruskal(G):
sort edges by weight
T = {}
for each v: makeSet(v)
for (u,v,w) in sorted edges:
if findSet(u) != findSet(v):
T = T + {(u,v,w)}
union(u, v)
return T
Prim(G, s):
for each v: key[v] = INF
key[s] = 0
PQ = all vertices
while PQ not empty:
u = extractMin(PQ)
for each neighbor v of u:
if w(u,v) < key[v]:
key[v] = w(u,v)
parent[v] = u
